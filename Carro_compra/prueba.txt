    <title>Eliminar Párrafo</title>

<body>
    <p id="parrafo">Párrafo 1</p>
    <p id="parrafo">Párrafo 2</p>
    <p id="parrafo">Párrafo 3</p>

    <script>
        // Selecciona todos los elementos con el mismo id
        const parrafos = document.querySelectorAll('#parrafo');

        // Añade un evento de clic a cada párrafo
        parrafos.forEach(parrafo => {
            parrafo.addEventListener('click', function() {
                // Elimina el párrafo clicado
                this.remove();
            });
        });
    </script>
</body>
